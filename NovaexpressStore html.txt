<script>
function calculatePrice() {
    const usd = parseFloat(document.getElementById("usdPrice").value);
    if (usd && usd > 0) {
        const priceDA = usd * 250; // Convert USD to DA
        const finalPrice = Math.round(priceDA * 1.25); // Add 25% and round
        document.getElementById("finalResult").innerHTML = 
            Final Price: ${finalPrice.toLocaleString()} DA;
    } else {
        document.getElementById("finalResult").innerHTML = "Please enter a valid amount.";
    }
}
</script>
