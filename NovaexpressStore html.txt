    // Products List - copy one block and edit name, price, img to add new products
    const PRODUCTS = [
      { id: 1, name: "üéí School Backpack", price: 2500, img: "https://i.ibb.co/0Y5V2Rk/backpack.jpg" },
      { id: 2, name: "‚úèÔ∏è Pencil Case", price: 800, img: "https://i.ibb.co/bQ08s5p/pencil-case.jpg" },
      { id: 3, name: "üìì Notebook Set", price: 1200, img: "https://i.ibb.co/SRKzLhY/notebook.jpg" },
      { id: 4, name: "ü•§ Water Bottle", price: 1500, img: "https://i.ibb.co/gZYc3Bk/water-bottle.jpg" }
    ];

    let cart = [];
    const productList = document.getElementById("product-list");
    const cartItems = document.getElementById("cart-items");
    const totalEl = document.getElementById("total");
    const cartDataInput = document.getElementById("cart-data");

    function renderProducts() {
      productList.innerHTML = PRODUCTS.map(p => `
        <div class="product">
          <img src="${p.img}" alt="${p.name}">
          <h3>${p.name}</h3>
          <p><strong>${p.price} DA</strong></p>
          <button onclick="addToCart(${p.id})">Add to Cart</button>
        </div>
      `).join("");
    }

    function addToCart(id) {
      const product = PRODUCTS.find(p => p.id === id);
      cart.push(product);
      renderCart();
    }

    function renderCart() {
      cartItems.innerHTML = cart.map(p => `<li>${p.name} - ${p.price} DA</li>`).join("");
      const total = cart.reduce((sum, p) => sum + p.price, 0);
      totalEl.textContent = total;
      cartDataInput.value = JSON.stringify(cart);
    }

    renderProducts();
